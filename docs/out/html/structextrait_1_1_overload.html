Detailed Description

template<class Signature, class = void>
struct extrait::Overload< Signature, class >

Used to disambiguate function pointers that point to functions with one or more overloads.

If you have a function pointer with several different overloads, it doesn't suffice to just pass that pointer to a extrait::Function template, you will need to specify the signature for a specific overload:

using extrait::Function, extrait::Overload; // Simplify our access for illustration purposes
void test();
int test(int);
using FirstFunc = Function&lt;Overload&lt;void()>::of(test)&gt;; // Works, selects "void test();"
using SecondFunc = Function&lt;Overload&lt;int(int)>::of(test)&gt;; // Works, selects "int test(int);"
using Func = Function&lt;test&gt;; // Errors, which overload is meant?
// This also works for member functions of course
struct Test
{
void test();
void test(int);
void test(int) const; // different qualification
};
using FirstFunc = Function&lt;Overload&lt;void()&gt;::of(&Test::test)&gt;; // Works, selects "void test();"
using SecondFunc = Function&lt;Overload&lt;void(int)&gt;::of(&Test::test)&gt;; // Works, selects "int test(int);"
using SecondFuncButConst = Function&lt;Overload&lt;void(int) const&gt;::of(&Test::test)&gt;; // Works, selects "int test(int) const;"
using Func = Function&lt;&Test::test&gt;; // Again, doesn't work, we are not telling the compiler WHAT function we are actually asking for
